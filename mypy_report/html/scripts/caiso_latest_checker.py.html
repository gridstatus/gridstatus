import polars as pl

polars_df = pl.from_pandas(caiso.get_load(date=datetime.now()))
print(polars_df)

fuel_df = pl.from_pandas(caiso.get_fuel_mix(date=datetime.now()))

storage_df = pl.from_pandas(caiso.get_storage(date=datetime.now()))
